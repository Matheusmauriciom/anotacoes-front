Retorno

      
  //simples
function somar(a, b) {
    return a + b;
  }
  let resultado = somar(2, 3);
  console.log(resultado); // 5

    //retorno condicional
function obterMaiorNumero(a, b) {
    if (a > b) {
      return a;
    } else {
      return b;
    }
  }
  let maiorNumero = obterMaiorNumero(7, 4);
  console.log(maiorNumero); // 7

  //retorno objetos
function criarPessoa(nome, idade) {
    return {
      nome: nome,
      idade: idade
    };
  }
  let pessoa = criarPessoa("Matheus", 25);
  console.log(pessoa); // Saida: { nome: "Matheus", idade: 25 }

// retorno de arrays
function obterNumerosPares(n) {
  let numerosPares = [];
    for (var i = 0; i <= n; i++) {
      if (i % 2 === 0) {
        numerosPares.push(i);
      }
    }
    return numerosPares;
  }
  let pares = obterNumerosPares(10);
  console.log(pares); // Saida: [0, 2, 4, 6, 8, 10]

callbacks !

um callback é uma função que é passada como argumento para outra função e é executada posteriormente, geralmente quando um determinado evento ocorre ou quando um processo assíncrono é concluído. O objetivo principal de um callback é permitir a execução de código após uma determinada tarefa ter sido concluída.

function executarDepoisDoTempo(callback, tempo) { setTimeout(callback, tempo); } 
  
function meuCallback() {
  console.log('Callback executado!');
}

executarDepoisDoTempo(meuCallback, 2000);

Callbacks também são amplamente utilizados em JavaScript para lidar com processos assíncronos, como solicitações de rede. Por exemplo, ao fazer uma solicitação AJAX usando a API fetch, você pode fornecer um callback para ser executado quando a resposta da solicitação estiver disponível:

  
    fetch('https://api.example.com/data')
  .then(function(response) {
    // Callback executado quando a resposta é recebida
    return response.json();
  })
  .then(function(data) {
    // Callback executado quando os dados JSON são processados
    console.log(data);
  })
  .catch(function(error) {
    // Callback executado em caso de erro
    console.error('Erro:', error);
  });
