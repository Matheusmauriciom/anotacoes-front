/* OBJETOS {} (const) Objetos são variáveis com muitos valores dentro. 
Os valores dentro de um objeto são chamados de propriedades.
Os objetos podem ter métodos.
Um método é uma função dentro de uma propriedade.
*/
// Exemplo 01: Objeto Literal
function exemploObjetoLiteral() {
  const carro = {
    marca: "Ford",
    modelo: "Ka",
    ano: 2015,
    placa: "ABC-1234",
    buzina: function () {
      alert("Bii");
    },
    completo: function () {
      return "A marca é " + this.marca + " e o modelo é: " + this.modelo;
      // 'this' refere-se ao objeto 'carro', permitindo o acesso às propriedades 'marca' e 'modelo'
    },
  };
  console.log(carro.completo()); // Saída : A marca é Ford e o modelo é: Ka
}
exemploObjetoLiteral();

Objeto com Método

      
// Exemplo 02: Objeto com Método
function exemploObjetoComMetodo() {
  const produto = {
    nome: "caneta",
    qtd: 10,
    comprar: function (n) {
      console.log(this);
      if (n > this.qtd) {
        return "Quantidade não disponível";
      }
      this.qtd -= n;
      // 'this' refere-se ao objeto 'produto', permitindo o acesso à propriedade 'qtd'
    },
  };
  produto.comprar(2); 
  console.log(produto);
// { nome: 'caneta', qtd: 10, comprar: [Function: comprar] }
// { nome: 'caneta', qtd: 8, comprar: [Function: comprar] }
}
exemploObjetoComMetodo();

Objeto simples

      
function exemploObjetoSimples() {
  const pessoa = new Object();
  pessoa.nome = "Matheus";
  pessoa.idade = 100;
  console.log(pessoa); // { nome: 'Matheus', idade: 100 }
  console.log(pessoa.nome); // Matheus
  console.log(pessoa.idade); // 100
}
exemploObjetoSimples();

Loop for...in

      
function exemploLoopForIn() {
  const pessoa = {
    nome: "Maria",
    idade: 28,
    email: "maria@teste.net",
  };

  console.log(pessoa); // { nome: 'Maria', idade: 28, email: 'maria@teste.net' }

  for (let prop in pessoa) {
    console.log(prop);
    console.log(pessoa[prop]);
  }
  // nome
  // Maria
  // idade
  // 28
  // email
  // maria@teste.net
}
exemploLoopForIn();

// Exemplo 06: Objeto com propriedades e métodos
function exemploObjetoPropriedadesMetodos() {
  const pessoa = {
    nome: "Matheus",
    idade: 100,
    profissao: "Programador",
    ola: function () {
      console.log(
        "Olá, meu nome é " + this.nome + " e tenho " + this.idade + " anos."
      );
      // 'this' refere-se ao objeto 'pessoa', permitindo o acesso às propriedades 'nome' e 'idade'
    },
    mudarProfissao: function (novaProfissao) {
      this.profissao = novaProfissao;
      console.log("A profissão foi alterada para " + novaProfissao);
      // 'this' refere-se ao objeto 'pessoa', permitindo a atualização da propriedade 'profissao'
    },
  };

  pessoa.ola(); // Olá, meu nome é Matheus e tenho 25 anos.
  pessoa.mudarProfissao("Engenheiro de Software"); // A profissão foi alterada para Engenheiro de Software
}
exemploObjetoPropriedadesMetodos();

Adição e remoção de propriedades de objetos

      
function exemploAdicaoRemocaoPropriedadesObjetos() {
  const pessoa = {
    nome: "Matheus",
    idade: 25,
  };

  pessoa.profissao = "Programador"; // Adiciona uma nova propriedade(profissao)
  console.log(pessoa); // { nome: 'Matheus', idade: 25, profissao: 'Programador' }

  delete pessoa.idade; // Remove uma propriedade existente(idade)
  console.log(pessoa); // { nome: 'Matheus', profissao: 'Programador' }
}
exemploAdicaoRemocaoPropriedadesObjetos();
