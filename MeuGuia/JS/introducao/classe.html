/*  Uma classe é uma estrutura que define um tipo de objeto, especificando seus atributos e métodos. 
Ela serve como um modelo ou plano para criar objetos específicos, que são chamados de instâncias da classe. (fabrica, molde) */

class Carro {
    constructor(valor1, valor2, valor3){
        this.marca = valor1;
        this.modelo =valor2;
        this.ano = valor3;
    }
    buzina (){
        return this.modelo+" buzinou: fon"
    }
}

const uno = new Carro ("Fiat", "uno" ,2001);
const gol = new Carro ("Volks", "Gol", 2013);
console.log(uno.buzina()); // uno buzinou: fon
console.log(uno.ano); // 2001

class Pessoa {
    constructor(nome, idade, sexo){
        this.nome = nome;
        this.idade = idade;
        this.sexo =sexo;
    }
    descricao (){
      console.log(`meu nome é ${this.nome}, tenho ${this.idade} anos e meu sexo é ${this.sexo}`)
    }
}
const pessoa1 = new Pessoa ("Matheus", 100, "masculino");
pessoa1.descricao(); // Saída : meu nome é Matheus, tenho 100 anos e meu sexo é masculino
