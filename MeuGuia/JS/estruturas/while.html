While

    
//Exemplo 1: Imprimindo os números pares de 1 a 10:
let num = 1;
while (num <= 10) {
  if (num % 2 === 0) {
    console.log(num);
  }
  num++;
}


//Exemplo 2: Calculando o fatorial de um número:
let number = 5;
let factorial = 1;
let i = 1;

while (i <= number) {
  factorial *= i;
  i++;
}
console.log("O fatorial de " + number + " é: " + factorial);

Do-while

            
//Exemplo 1: Imprimindo números de 1 a 5 usando 
let i = 1;
do {
  console.log(i);
  i++;
} while (i <= 5);

// Exemplo 2 (PROMPT)
let password;
let correctPassword = "1234";

do {
  password = prompt("Digite a senha:");
} while (password !== correctPassword);

console.log("Acesso concedido!");
