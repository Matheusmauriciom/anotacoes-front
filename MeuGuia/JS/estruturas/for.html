Estruturas de repetição

  • Utilização de arrays para armazenar e manipular conjuntos de dados
  • Utilização de loops (for, while) para percorrer e processar elementos de um array
  • Criação e manipulação de arrays em JavaScript
  • Utilização de métodos de array, como push, pop, splice, map, filter e reduce
  • Iteração sobre os elementos de um array usando loops

For

      
//contagem crescente
for(let i=0; i <= 5; i++){
    console.log(i);
}

//contagem descrescente
for(let i=10; i >= 1; i--){
    console.log(i);
}

// aumentando o valor de i em 2 acada iteração
for (let i = 0; i <= 10; i += 2) {
  console.log(i);
}

Loop em objetos

        
//loop em objetos
const carro = {
  marca: "Toyota",
  modelo: "Corolla",
  ano: 2022,
};
for (let chave in carro) {
  console.log(chave + ": " + carro[chave]);
}

//FOR OBJETO
const carros = ["Gol", "Corsa", "Uno", "Fiesta"];
var tamanho = carros.length;

for (let i = 0; i < tamanho; i++) {
  console.log(carros[i] + " - ");
}

Array em objetos

        
//Loop com array de objetos
const pessoas = [
  { nome: 'João', idade: 25 },
  { nome: 'Maria', idade: 30 },
  { nome: 'Pedro', idade: 28 }
];
for (let pessoa of pessoas) {
  console.log(pessoa.nome + ' tem ' + pessoa.idade + ' anos.');
}

//iteração em um array
let frutas =["maça","banana","laranja"];
for( let i=0; i < frutas.length; i++){
    console.log(frutas[i]);
}

Break / Continue

        
//continue
for(let i =1; i <=10; i++){
    if(i % 2 === 0){
        continue; // pula iterações pares
    }
    console.log(i);
}

//break
for(let i =1; i <=10; i++){
    if(i  === 6){
        break; // pula iterações pares
    }
    console.log(i);
}
